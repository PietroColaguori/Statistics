// While loop

{
    // Code to execute
}

{

// Foreach loop

}

// Break statement
for (int i = 0; i < 10; i++)
{

        break;
    }
    // Code to execute
}

// Continue statement

{
    if (i == 5)

        continue;
    }
    // Code to execute
}
